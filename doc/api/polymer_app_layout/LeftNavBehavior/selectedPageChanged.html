@Observe("selectedPage")
selectedPageChanged(Page newValue) {
  if (nav.parent != null) {
    if (newValue.hideLeftNav) {
      nav.parent.style.setProperty("display", "none");
    } else {
      nav.parent.style.setProperty("display", "block");
    }
  }
}
